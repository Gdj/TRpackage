clipboard

clipboard 플러그인을 이용해 문자열을 클립보드에 저장합니다.

zclip( {options} )

설명
내용르 클립보드에 저장합니다.
zclip 설명
속성 이름 속성 설명
path ZeroClipboard.swf 파일 위치
copy 클립 보드에 들어갈 내용.
afterCopy 알럿창 내용을 입력합니다. 설정하지 않으면 클립보드 내용이 들어갑니다.

	<script type="text/javascript" src="http://ajax.googleapis.com/ajax/libs/jquery/1.11.1/jquery.min.js" ></script>

</head>

<body>

	<textarea id="text-to-copy" cols="50" >./copy.html 파일내용이 클리보드에 저장됩니다. </textarea>
	<br />
	<button id="copy-button"  title="복사하기">복사하기</button> <br /><br />
	
	<textarea id="" cols="50" rows="30"  onfocus="this.value='';" ></textarea> 

<script type="text/javascript">
$(document).ready(function(){



	
	// 클립보드 
	function copyURL($btn, $url){ 
		
		var request = $.ajax({
		  url: $url,
		  method: "GET",			
		  dataType: "html"
		});
		request.done(function( msg ) {			 
		  var _data = msg ;
		  $($btn).attr("data-clipboard-text", _data );
			// 클립보드 선언


				$("#copy-button").zclip({
					path: "js/ZeroClipboard.swf",
					copy: $( "#text-to-copy" ).attr("data-clipboard-text") ,
					afterCopy:function(){ alert("복사 되었습니다.") }

				});



		});

		// 실패했을경우
		request.fail(function( jqXHR, textStatus ) {
		  alert( jqXHR + ": Request failed: " + textStatus );
		});		

	}

		
	copyURL( "#text-to-copy" ,"./copy.html");	


});
</script>


</body>
</html>
