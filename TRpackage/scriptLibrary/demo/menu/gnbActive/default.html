GNB active

GNB Active 네비게이션
2뎁스까지 GNB 메뉴 활성화 가능 하도록 한메뉴.

.trpGNBactive( $activeClass:string, $activeFn:function )

설명
기본 위치값은 CSS 에서 설정해야합니다.
GNB active 설명
매개변수 매개변수 설명
$activeClass 메뉴의 활성화 클래스 이름
기본값은 "on"클래스로 되여있습니다.
$activeFn($depth1, $depth2 ) 1뎁스, 2뎁스 리턴함수에 파라미터 값을 담아서 리턴.
gnb가 실행될때마다 콜백리턴 함수를 호출해 확장성있게 활용할수있습니다.
tab menu 설명
메서드 메서드 설명
setBtn(index | array) 인덱스번호로 gnb메뉴를 활성화 합니다.
2뎁스 gnb의 경우 배열 형태로 활성화 값을 줄수있습니다.

      var _oldDep1=-1;
      function gnbActiave($dep1, $dep2){
        var _dep1_ul = $('#gnb').find(' > ul ');							//  뎁스 1 ul
        var _dep2_ul = $('#gnb').find(' > ul ul');							//  뎁스 2 ul
        if($dep1 == -1){													//	1뎁스가 없으면 초기화
          _dep1_ul.find('li > a').removeClass("active");
          _dep2_ul.css( {'display':'none'} );
          _dep2_ul.eq($dep1).find(' li a').removeClass("active");
          return;
        }

        _dep1_ul.find(' li > a').removeClass("active");						//뎁스1	숨김
        _dep1_ul.find(' > li').eq($dep1).find(' > a').addClass("active");	//뎁스1	보임

        if(_oldDep1 != $dep1 ){												//뎁스2	숨김
            _dep2_ul.css({'display':'none'});						
        }	
        _dep2_ul.eq($dep1).css({'display':'block'});						//뎁스2	보임
        //_dep2_ul.eq($dep1).stop().slideDown();			

        if($dep2 != -1){													//뎁스2 가없을때
         _dep2_ul.eq($dep1).find(' li a').removeClass("active");
         _dep2_ul.eq($dep1).find(' li:eq('+($dep2)+') a').addClass("active");		  
        }
        _oldDep1 = $dep1;
        return false;			// 내부정의 사용하지 않음
      }
    })
