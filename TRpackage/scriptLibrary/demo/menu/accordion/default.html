accordion menu

선택된 메뉴만 활성화되는 아코디언 메뉴

.trpAccordionMenu(button:string , contant:string, activeClass:string , activeFN:function )

설명
기본 위치값은 CSS 에서 설정해야합니다.
Accordion menu 설명
매개변수 매개변수 설명
button 아코디언 메뉴 번튼 선택자
contant 아코디언 메뉴의 컨텐츠 선택자
activeClass 메뉴의 활성화 클래스 이름
기본값은 "on"클래스로 되여있습니다.
activeFN($index) 선택된 인덱스를 넘김니다.
"return false;" 값을넘기면 기본실행되는 스크립트가 실행하지 않고 custom 으로 제작할수있습니다. 예제2)
tab menu 설명
메서드 메서드 설명
setBtn(index) 인덱스번호로 아코디언 메뉴를 활성화 합니다.

    $(document).ready(function(){

      $("#accordion").trpAccordionMenu(".acc a" ,".ac_con" , "on" , activeFn).setBtn(1);
      function activeFn($index){
        if( $("#accordion").find(".acc a").eq($index).hasClass("on") ){ 
            $("#accordion").find(".acc a").eq($index).removeClass("on");
            $("#accordion").find(".ac_con").eq($index).slideUp();
            return ;
        } // 활성화되어있으면 닫는다

        $("#accordion").find(".acc a").removeClass("on");
        $("#accordion").find(".acc a").eq($index).addClass("on");

        $("#accordion").find(".ac_con").slideUp();
        $("#accordion").find(".ac_con").eq($index).slideDown();
          return false;
      }

    })
  </script>
