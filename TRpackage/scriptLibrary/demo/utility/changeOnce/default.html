trpUtilityChangeOnce

trpUtilityChangeOnce($value, $activeFn)

설명
trpUtility.js 불러오면 trpBrowser를 사용할수 있습니다.

class 설명
파라미터 이름 파마미너 설명
$value 변경되는 값
$activeFn 값이 바뀔때 실행될 함수

        $(window).resize(function(){
            var _change_width = $(window).width();
            var _temp_width;
            if( _change_width >= 1204){
                _temp_width = "width_max";
            }else if( _change_width >= 760 ){
                _temp_width = "width_big";
            }else if( _change_width >= 320 ){
                _temp_width = "width_middle";
            }else{
                _temp_width = "width_small";
            };

            trpUtilityChangeOnce(_temp_width, function($chang){
                $("body").removeClass("width_max width_big width_middle width_small");
                $("body").addClass($chang);
                $("h2").text($chang);
            });

        })
