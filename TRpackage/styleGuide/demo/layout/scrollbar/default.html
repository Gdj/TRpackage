  /*  ie 용 스크롤 바  *********************************************/
  .trp_scrollbar_y { 
    overflow-y:auto; 
    border: 1px solid #333; padding: 10px; width: 100px; height: 150px; 
    scrollbar-arrow-color: #ffffff; 
    scrollbar-Face-Color: #dadada;
    scrollbar-Track-Color: #ffffff;
    scrollbar-base-color: #ffffff;
    scrollbar-3dLight-Color: #f3f3f3;
    scrollbar-DarkShadow-Color: #f3f3f3;
    scrollbar-Highlight-Color: #f3f3f3;
    scrollbar-Shadow-Color: #f3f3f3;
  }
  /* webkit (Chrome, Safari, firefox )용 스크롤 바 */
  .trp_scrollbar_y::-webkit-scrollbar {width: 15px;}
  .trp_scrollbar_y::-webkit-scrollbar-button:start:decrement,
  .trp_scrollbar_y::-webkit-scrollbar-button:end:increment { display:block; width:0px; height:0px; }
  .trp_scrollbar_y::-webkit-scrollbar-thumb { background: rgba(0,0,0,.1); }

</style>
