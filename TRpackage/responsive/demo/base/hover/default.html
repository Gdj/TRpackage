      font-size: 30px;  color:#fff; border: 0; background-color:#330000; 
      text-align: center; vertical-align: middle; line-height: 100px;

    /* hover가 가능할 때만 적용 */
    @media (hover: hover) {
      .pc_hover:hover {
        background-color:#33ff00;
      }

2. css & javascript로 대응
모바일 디바이스 체크후 pc일때 body에 "device_pc"클래스 추가
"device_pc" 클래스가 있을대만 hover 적용

  <script>
    $(window).ready(function(){
      // 접속 디바이스를 확인합니다(데스크탑:false, 모바일:true) 
      function isTouchAble(){ 
        if(navigator.userAgent.indexOf('Mobile') != -1){ 
          alert("===mobile===")
          return true; 
        } else{ 
          alert("===pc===")
          $("body").addClass("device_pc");
          return false; 
        } 
      }
      isTouchAble();
    });
  </script>
