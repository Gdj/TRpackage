  <style>
     .table-wrap { position:relative; border-top: 2px solid #000; }
     .trp.table-data-type1 { border-collapse: initial; width: 100%; border-top: 0; }
     .trp.table-data-type1 thead th, .trp.table-data-type1 thead td,
     .trp.table-data-type1 tbody th, .trp.table-data-type1 tbody td { height: 50px; box-sizing: border-box;}
     .table--min table { min-width:700px; }
     .table-box.table--horizontal th { width:15%;  }  
     .mb_layout_t { text-align: center;}
 
     @media screen and (max-width:768px) {
       /* 테이블 스타일 */
       .mb_layout_t{ display: table; table-layout: fixed; border-collapse: collapse; width: 100%; height: 100%;}
       .mb_layout_t .layout_r{ display: table-row;}
       .mb_layout_t .layout_c{ display: table-cell; vertical-align: middle; text-align: center;}
 
       .trp.table-data-type1 { width: auto; }
 
       /*horizontal*/
       .table-box.table--horizontal   { overflow-x:auto; overflow-y: hidden; }
       .table-box.table--horizontal table  { padding-left: 80px; }
       .table-box.table--horizontal tr { position:relative; }
       .table-box.table--horizontal th { position:fixed; width:80px; margin-left: -80px; }  
       .table-box.table--horizontal th + td {  }  
     }
   </style>
