.mediaQuery:has(.c1) .c1 { border: 3px solid #3f3; }
.mediaQuery:has(.c2) .c2 { border: 3px solid #d35; }
.mediaQuery:has(.c3) .c3 { border: 3px solid #777; }
.mediaQuery:has(.c4) .c4 { border: 3px solid #369; }


/* 화면 방향 : 가로 | 세로 [landscape | portrait] 
*/
@media (orientation: landscape) {
	.mediaQuery .c1 { background-color: #3f3; }
}

/* 해상도 : 레티나 해상도 */
@media (min-resolution: 2dppx) {
	.mediaQuery .c2 { background-color: #d35; }
}

/* 시스템 설정 모드 : [light | dark] */
@media (prefers-color-scheme: dark) {
	.mediaQuery .c3{ background-color: #777; }
}

/* hover 지원 : pc, mobile[hover, none]*/
@media (hover: hover) {
	.mediaQuery .c4 { background-color: #369 }
}
