trpUtilityChangeOnceFn

같은 값으로 호출되면 1번만 실행되고, 다른 값이 들어 올때만 실행되는 함수 구연

trpUtilityChangeOnceFn($functionChange)

설명
trpUtility.js 불러오면 trpUtilityChangeOnceFn 사용할수 있습니다.
<script type="text/javascript" src="./js/libs/trpUtility.js"></script>
class 설명
파라미터 이름 파라미터 설명
$functionChange 같은값이 들어와도 한번만 실행될 함수
class 설명
메서드 이름 메서드 설명
setChange($string) $string 변화된 문자열을 입력시 함수가 실행된다.

	$(document).ready(function(){
    var _classNames = ["is_pc", "is_tb", "is_mb"]   
    var _className  = "";
    var _win_width ="";
    var onceFn = $(".width_check").trpUtilityChangeOnceFn(
      function(){ 
        $(".width_check").removeClass("is_pc is_tb is_mb");
        $(".width_check").addClass(_className);     
        $(".width_check .p1").text("width 1200 / 768 / 768 under : " + _className);     
        $(".width_check .p2").text( "width : " + _win_width );
        console.log(" width : " + _win_width ) ;
      } 
    );
    $(window).resize(function(){ 
      _win_width = window.innerWidth; //  $(window).innerWidth();
      if( _win_width > 1200){
        _className = _classNames[0];
      }else if( _win_width > 768 ){
        _className = _classNames[1];
      }else{
        _className = _classNames[2];
      }
      onceFn.setChange(_className);
    });
    $(window).trigger("resize");

	})
	</script>

    setInterval(function(){
      ++_count2;

      console.log(_count2);

      onceFn2.setChange( _changed );
    }, 500);
