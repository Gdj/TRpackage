trpGallery

기본적인 슬라이딩 겔러리입니다.
겔러리의 기본로직만 제공되어 확장성을 높였습니다.
일반적인 플러그인보다 사용방법이 다소 번거러울수 있습니다.

trpGallery($wrapSelector, $itemSelector)

설명
기본 위치값은 CSS 에서 설정해야합니다.
trpGallery 파라미터 설명
trpGallery 매개변수 trpGallery 매개변수 설명
$wrapSelector 팝업을 감싸고 있는 최종 컨테이너 선택자!(필수)
$itemSelector 팝업을 보여주게 될 팝업 컨텐츠(필수)
trpGallery 속성
trpGallery 속성 trpGallery 속성 설명
.index 현제 진행되는 모션의 인덱스 값을 담고 있습니다.
trpGallery methods
trpGallery 메서드 trpGallery 메서드 설명
.motionEnd($index , $viewNum) 스라이딩 모션이 끝에 도달하게 더이상 진행하지 않습니다.
$index 값으로 이동합니다. (필수)
$viewNum 화면에 보여지는 겟수를 표기합니다. 내부로직만 영향을 받습니다. (입력이 안되면 1이 기본 값입니다.)
.motionReturn($index , $viewNum) 스라이딩 모션이 끝에 도달하게 더이상 진행하지 않습니다.
$index값으로 이동합니다. (필수)
$viewNum 화면에 보여지는 겟수를 표기합니다. 내부로직만 영향을 받습니다. (입력이 안되면 1이 기본 값입니다.)
.motionLoop($index) 스라이딩 모션이 회전하듯이 게속돌게됩니다.
$index값으로 이동합니다.

<script type="text/javascript">
	
	
		$(document).ready(function(){
			
			/*================  갤러리 생성 =================== #thumbNailView >  */					
			var galeryMain 	= new trpGallery(".imgallery-wrapper"		,".imgallery-wrapper .item");
			var _total= $(".imgallery-wrapper .item").length  ;			// 리스트 겟수
			var _widthAll = _total*400;
			var _aotoPlay = true;
			$(".imgallery-wrapper").css({ width:_widthAll });
			

			for(var $i=0 ; $i<_total ; ++$i ){			
				$(".pagination ul").prepend( "<li class='dots'><a href='#'></a></li>" );
			}
			$(".pagination ul li.dots").eq(0).addClass("on");
			
			/*================  화살 버튼  =================== */
			$(".gallery_view .prev").on("click", function(){ 				
				timerStart();
				activeFn( (galeryMain.index -1)  );
			})	
			$(".gallery_view .next").on("click", function(){
				timerStart();
				activeFn( (galeryMain.index +1) );					
			})

			/*================  dot 버튼  =================== */ 		
			$( ".pagination ul" ).on( "click","a",function($e,$name){				
				activeFn( $(this).parent().index() );				
			})
			
			/*================  겔러리실행  =================== */ 
			function activeFn($index){				
				galeryMain.motionLoop( $index );				
				$(".pagination ul li.dots").removeClass("on");
				$(".pagination ul li.dots").eq(galeryMain.index).addClass("on");

				_aotoPlay = true;				
			}

			/*================  타이머   =================== */
			
			var _count=0; 			// 인덱스
			var _timerB=true;		// 타이머 체크
			var _timer;				// 타이머		
			
			function timerStart(){					
				if(!_timerB)return;
				timerStop();
				if(!_timer) _timer = setInterval( loofFn ,5000);
			}
			function timerStop(){
				if(_timer){
					clearInterval(_timer);
					_timer = null;
				}
			}
			function loofFn(){					
				activeFn(galeryMain.index +1);
			}

			timerStart(); //타이머 시작
				
		}) // ready
		
	

    </script>
