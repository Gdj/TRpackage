vimeo API

vimeo API를 이용한 컨트롤

Vimeo.Player( id or url ,{ option } )

설명
기본 위치값은 CSS 에서 설정해야합니다.
버튼 옵션 정의
option
매개변수 매개변수 설명
width 동영상 플레이어의 너비입니다. 기본값은 480입니다.
height 동영상 플레이어의 높이입니다. 기본값은 270입니다.
id 플레이어에서 로드할 동영상을 식별하는 Vimeo 동영상 ID입니다
autoplay 기본값은 false 이며 true 설정시 자동 플레이 됩니다.
메서드
메서드 메서드 설명
.play() 현재 신호를 준 동영상과 로드한 동영상을 재생합니다. 이 함수를 실행한 후 플레이어의 마지막 상태는 playing(1)입니다.
.pause() 현재 재생 중인 동영상을 일시중지합니다. 함수가 호출될 때 플레이어 상태가 변경되지 않는 ended(0) 상태에 있지 않는 한 이 함수를 실행한 후 플레이어의 마지막 상태는 paused(2)입니다.
.setCurrentTime(seconds:Number) 지정한 초로 플레이 해드 이동
.getCurrentTime() 현제 플레이 해드를 가져옵니다.
unload() 동영상을 초기상태로 돌립니다.
setVolume(volume:Number) 볼륨을 설정합니다. 0에서 1 사이의 수를 허용합니다.
getVolume() 플레이어의 현재 볼륨을 0에서 100 사이의 정수로 반환합니다.
event
이벤트 이벤트 설명
.on(event, function(data){ ... }) event("play", "pause", "ended", "progress", "seeked", "error", "loaded" ..)
  • data.duration : 재생시간 초
  • data.percent : 진행 퍼센트 (1=100%)
  • data.seconds : 현제초
event

<!doctype html>
<html lang="en">
<head>
	<meta charset="UTF-8">
	<title>vimeo player</title>
	<script src="https://player.vimeo.com/api/player.js"></script>
</head>
<body>

	<!-- video -->
	<div id="made-in-ny"></div>

	<!-- <iframe id="vimeoVideo" src="https://player.vimeo.com/video/59777392?autoplay=0&color=ffffff&byline=0&portrait=0" width="100%" height="100%" frameborder="0" webkitallowfullscreen mozallowfullscreen allowfullscreen></iframe> -->
	<!-- //video -->
	
	<div class="menu">
		<ul>
			<li><a href="javascript:void(0)" onclick="playVideo()">재생</a></li>
			<li><a href="javascript:void(0)" onclick="stopVideo()">정지</a></li>
			<li><a href="javascript:void(0)" onclick="pauseVideo()">일시정지</a></li>
			<li><a href="javascript:void(0)" onclick="seekTo(50)">50초뒤로 이동</a></li>
			<li><a href="javascript:void(0)" onclick="setVolume(0.5)">볼룸 0.5</a></li>			
		</ul>
	</div>

	<script>
		// Create a Player
		var options = { id: 59777392, width: 640, loop: false, autoplay:true  };
		var vimeoPlayer = new Vimeo.Player('made-in-ny', options);



		//var vimeoPlayer =  new Vimeo.Player( $("#vimeoVideo") );   //iframe 
			


	  /* 재생 */
      function playVideo() {
		vimeoPlayer.play();
      }

	  /* 재생정지 */
      function stopVideo() {
		vimeoPlayer.setCurrentTime(0);
		vimeoPlayer.pause();
      }

	  /* 재생 일시정지*/
	  function pauseVideo() {
        vimeoPlayer.pause();
      }

	  /* seekTo */
	  function seekTo($seconds ) {
		vimeoPlayer.setCurrentTime($seconds);
      }
		

	   /* 볼륨 조절 : 볼륨을 설정합니다. 0에서 1 사이의 수를 허용합니다. */
      function setVolume($volume) {
        vimeoPlayer.setVolume($volume);
      }
	  

      /* event */
	  vimeoPlayer.on('ended', function(data) {
			console.log( "재생끝 " +  data.duration, data.percent, data.seconds);
	  });


	</script>
</body>
</html>
