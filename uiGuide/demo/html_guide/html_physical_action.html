<script>
	function calculate() {
		const num1 = parseFloat(document.getElementById('standard-input').value) || 0;
		const num2 = parseFloat(document.getElementById('variation-input').value) || 0;
		document.getElementById('calculator-output').textContent = Math.round((num2 / num1 * 100) - 100);
	}
	calculate();
</script>

<script>
	// 현재 날짜로 date input 초기화
	const today = new Date();
	const currentYear = today.getFullYear();
	const currentMonth = String(today.getMonth() + 1).padStart(2, '0');
	const currentDay = String(today.getDate()).padStart(2, '0');
	const currentDateString = `${currentYear}-${currentMonth}-${currentDay}`;
	document.getElementById('date-current').value = currentDateString;

	// 날짜 입력 유효성 검사
	document.getElementById('date-current').addEventListener('change', function() {
		const selectedDate = new Date(this.value);
		const year = selectedDate.getFullYear();
		if (year < 1900 || year > 2099) {
			alert('연도는 1900년부터 9999년 사이여야 합니다.');
			this.value = '';
		}
	});
</script>
